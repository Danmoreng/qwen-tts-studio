import React, { useState, useEffect } from 'react';
import { Play, Square, Settings, Mic, Upload, Download, Sparkles, Volume2, Plus, Trash2, FileAudio, Moon, Sun, Activity, Smile, VolumeX, Zap } from 'lucide-react';

export default function App() {
  const [activeTab, setActiveTab] = useState('synthesizer');
  const [text, setText] = useState('');
  const [isGenerating, setIsGenerating] = useState(false);
  const [selectedVoice, setSelectedVoice] = useState('Default Voice (EN)');
  const [isDarkMode, setIsDarkMode] = useState(true); // Let's start directly in the cool dark mode
  const [selectedEmotion, setSelectedEmotion] = useState('neutral');

  // Mock data for cloned voices
  const [voices, setVoices] = useState([
    { id: 1, name: 'Default Voice (EN)', type: 'system' },
    { id: 2, name: 'My Voice (Clone)', type: 'cloned' }
  ]);

  const emotions = [
    { id: 'neutral', label: 'Neutral', icon: <Volume2 size={14} /> },
    { id: 'happy', label: 'Happy', icon: <Smile size={14} /> },
    { id: 'whisper', label: 'Whisper', icon: <VolumeX size={14} /> },
    { id: 'dynamic', label: 'Dynamic', icon: <Zap size={14} /> },
  ];

  const handleGenerate = () => {
    if (!text.trim()) return;
    setIsGenerating(true);
    setTimeout(() => setIsGenerating(false), 2500);
  };

  return (
    <div className={`${isDarkMode ? 'dark' : ''} h-screen w-full`}>
      <div className="flex h-full w-full bg-[#FAFAFA] dark:bg-[#0E1117] text-gray-800 dark:text-gray-200 font-sans transition-colors duration-300">
        
        {/* Navigation Rail */}
        <div className="w-20 bg-white dark:bg-[#161B22] border-r border-gray-200 dark:border-gray-800 flex flex-col items-center py-6 shadow-sm z-10 transition-colors duration-300">
          <div className="w-12 h-12 bg-blue-600 rounded-xl flex items-center justify-center text-white font-bold text-xl shadow-md mb-8">
            Q3
          </div>
          
          <nav className="flex flex-col gap-4 w-full px-2 flex-1">
            <NavItem icon={<Volume2 size={24} />} label="Studio" isActive={activeTab === 'synthesizer'} onClick={() => setActiveTab('synthesizer')} />
            <NavItem icon={<Mic size={24} />} label="Voices" isActive={activeTab === 'voices'} onClick={() => setActiveTab('voices')} />
            <NavItem icon={<Settings size={24} />} label="Setup" isActive={activeTab === 'settings'} onClick={() => setActiveTab('settings')} />
          </nav>

          {/* Theme Toggle */}
          <button 
            onClick={() => setIsDarkMode(!isDarkMode)}
            className="p-3 text-gray-400 hover:text-blue-500 dark:hover:text-blue-400 transition-colors rounded-xl hover:bg-gray-50 dark:hover:bg-gray-800"
            title="Toggle Dark/Light Mode"
          >
            {isDarkMode ? <Sun size={24} /> : <Moon size={24} />}
          </button>
        </div>

        {/* Main Content Area */}
        <div className="flex-1 flex flex-col h-full overflow-hidden relative">
          
          {/* Header */}
          <header className="h-16 px-8 flex items-center justify-between border-b border-gray-100 dark:border-gray-800 bg-white dark:bg-[#161B22] transition-colors duration-300">
            <h1 className="text-xl font-semibold text-gray-800 dark:text-white">
              {activeTab === 'synthesizer' && 'Speech Synthesis'}
              {activeTab === 'voices' && 'Voice Cloning'}
              {activeTab === 'settings' && 'Model Settings'}
            </h1>
            
            <div className="flex items-center gap-4">
              {/* Hardware Monitor */}
              <div className="hidden md:flex items-center gap-3 text-xs font-mono text-gray-500 dark:text-gray-400 bg-gray-50 dark:bg-[#0E1117] px-3 py-1.5 rounded-lg border border-gray-200 dark:border-gray-800">
                <Activity size={14} className="text-blue-500" />
                <span>RAM: 2.4 GB</span>
                <span className="w-px h-3 bg-gray-300 dark:bg-gray-700"></span>
                <span>CPU: {isGenerating ? '84%' : '2%'}</span>
              </div>

              {/* Status Indicator */}
              <div className="flex items-center gap-2 text-sm text-gray-600 dark:text-gray-300 bg-green-50 dark:bg-green-900/20 px-3 py-1.5 rounded-full border border-green-200 dark:border-green-800/50">
                <div className={`w-2 h-2 rounded-full ${isGenerating ? 'bg-amber-500 animate-pulse' : 'bg-green-500'}`}></div>
                {isGenerating ? 'Generating...' : 'Engine Ready'}
              </div>
            </div>
          </header>

          {/* Dynamic Content */}
          <main className="flex-1 overflow-y-auto p-8">
            
            {/* SYNTHESIZER VIEW */}
            {activeTab === 'synthesizer' && (
              <div className="max-w-4xl mx-auto h-full flex flex-col gap-6">
                
                {/* Controls Area */}
                <div className="flex flex-col md:flex-row justify-between items-start md:items-center gap-4 bg-white dark:bg-[#161B22] p-4 rounded-2xl border border-gray-200 dark:border-gray-800 shadow-sm transition-colors duration-300">
                  <div className="flex items-center gap-3 w-full md:w-auto">
                    <div className="w-10 h-10 rounded-full bg-blue-100 dark:bg-blue-900/30 flex items-center justify-center text-blue-600 dark:text-blue-400 shrink-0">
                      <Mic size={20} />
                    </div>
                    <select 
                      value={selectedVoice}
                      onChange={(e) => setSelectedVoice(e.target.value)}
                      className="bg-transparent border-none text-gray-900 dark:text-white font-medium text-base focus:ring-0 outline-none cursor-pointer w-full md:w-48"
                    >
                      {voices.map(v => (
                        <option key={v.id} value={v.name} className="dark:bg-gray-800">{v.name}</option>
                      ))}
                    </select>
                  </div>
                  
                  {/* Emotion/Style Selector */}
                  <div className="flex flex-wrap gap-2">
                    {emotions.map(emo => (
                      <button
                        key={emo.id}
                        onClick={() => setSelectedEmotion(emo.id)}
                        className={`flex items-center gap-1.5 px-3 py-1.5 rounded-full text-xs font-medium transition-all ${
                          selectedEmotion === emo.id 
                            ? 'bg-blue-100 text-blue-700 dark:bg-blue-900/40 dark:text-blue-300 border border-blue-200 dark:border-blue-800/50' 
                            : 'bg-gray-100 text-gray-600 dark:bg-gray-800 dark:text-gray-400 border border-transparent hover:bg-gray-200 dark:hover:bg-gray-700'
                        }`}
                      >
                        {emo.icon} {emo.label}
                      </button>
                    ))}
                  </div>
                </div>

                {/* Text Input Area */}
                <div className="flex-1 relative bg-white dark:bg-[#161B22] rounded-2xl shadow-sm border border-gray-200 dark:border-gray-800 overflow-hidden flex flex-col transition-colors duration-300">
                  <textarea
                    value={text}
                    onChange={(e) => setText(e.target.value)}
                    placeholder="Enter the text you want to be spoken here..."
                    className="flex-1 w-full p-6 text-lg resize-none outline-none text-gray-800 dark:text-gray-200 placeholder-gray-400 dark:placeholder-gray-600 bg-transparent leading-relaxed"
                  ></textarea>
                  
                  <div className="bg-gray-50 dark:bg-[#0E1117] border-t border-gray-100 dark:border-gray-800 p-4 flex justify-between items-center transition-colors duration-300">
                    <div className="text-sm text-gray-400 dark:text-gray-500 font-mono">
                      {text.length} / 5000 characters
                    </div>
                    
                    <button 
                      onClick={handleGenerate}
                      disabled={text.length === 0 || isGenerating}
                      className={`flex items-center gap-2 px-6 py-2.5 rounded-full font-medium transition-all ${
                        text.length === 0 
                          ? 'bg-gray-200 text-gray-400 dark:bg-gray-800 dark:text-gray-600 cursor-not-allowed'
                          : 'bg-blue-600 hover:bg-blue-700 text-white shadow-md hover:shadow-lg dark:shadow-blue-900/20'
                      }`}
                    >
                      {isGenerating ? (
                        <><Sparkles size={18} className="animate-pulse" /> Processing...</>
                      ) : (
                        <><Play size={18} fill="currentColor" /> Read Aloud</>
                      )}
                    </button>
                  </div>
                </div>

                {/* Studio Player (Waveform Mockup) */}
                <div className={`h-24 bg-white dark:bg-[#161B22] rounded-2xl shadow-sm border ${isGenerating ? 'border-blue-400 dark:border-blue-600 shadow-blue-100 dark:shadow-blue-900/20' : 'border-gray-200 dark:border-gray-800'} flex items-center px-6 gap-6 transition-all duration-300`}>
                   <button className="w-12 h-12 shrink-0 rounded-full bg-blue-50 dark:bg-blue-900/20 flex items-center justify-center text-blue-600 dark:text-blue-400 hover:bg-blue-100 dark:hover:bg-blue-900/40 transition-colors">
                      {isGenerating ? <Square size={20} fill="currentColor" className="animate-pulse" /> : <Play size={20} fill="currentColor" className="ml-1" />}
                   </button>
                   
                   {/* Waveform Visualization */}
                   <div className="flex-1 h-10 flex items-center gap-[2px] overflow-hidden">
                      {[...Array(60)].map((_, i) => (
                        <div 
                          key={i} 
                          className={`w-1.5 rounded-full transition-all duration-200 ${
                            i < 20 && !isGenerating ? 'bg-blue-500 dark:bg-blue-400' : 
                            isGenerating ? 'bg-blue-300 dark:bg-blue-800 animate-pulse' : 'bg-gray-200 dark:bg-gray-700'
                          }`}
                          style={{ 
                            height: isGenerating ? `${Math.random() * 100}%` : `${20 + Math.sin(i * 0.5) * 40 + Math.random() * 20}%`,
                            animationDelay: `${i * 0.05}s`
                          }}
                        ></div>
                      ))}
                   </div>

                   <div className="text-sm font-mono text-gray-500 dark:text-gray-400 shrink-0">0:04 / 0:12</div>
                   <button className="p-2 text-gray-400 hover:text-blue-600 dark:hover:text-blue-400 shrink-0 transition-colors" title="Export (.wav)">
                      <Download size={22} />
                   </button>
                </div>

              </div>
            )}

            {/* VOICES & SETTINGS remain largely similar, updated with dark mode classes */}
            {/* VOICES (CLONING) VIEW */}
            {activeTab === 'voices' && (
              <div className="max-w-4xl mx-auto flex flex-col gap-8">
                <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-6">
                  <div className="bg-blue-50 dark:bg-blue-900/10 border-2 border-dashed border-blue-200 dark:border-blue-800/50 rounded-2xl p-6 flex flex-col items-center justify-center text-center cursor-pointer hover:bg-blue-100 dark:hover:bg-blue-900/20 transition-colors min-h-[200px] group">
                    <div className="w-12 h-12 rounded-full bg-blue-100 dark:bg-blue-900/40 text-blue-600 dark:text-blue-400 flex items-center justify-center mb-4 group-hover:scale-110 transition-transform">
                      <Plus size={24} />
                    </div>
                    <h3 className="font-semibold text-blue-900 dark:text-blue-300">Clone New Voice</h3>
                    <p className="text-sm text-blue-600 dark:text-blue-400/70 mt-2">Upload a 10-30s audio sample (.wav)</p>
                  </div>

                  {voices.map(voice => (
                    <div key={voice.id} className="bg-white dark:bg-[#161B22] border border-gray-200 dark:border-gray-800 rounded-2xl p-6 flex flex-col relative shadow-sm hover:shadow-md dark:shadow-none transition-shadow min-h-[200px]">
                      {voice.type === 'system' ? (
                        <div className="absolute top-4 right-4 bg-gray-100 dark:bg-gray-800 text-gray-500 dark:text-gray-400 text-xs px-2 py-1 rounded font-medium">System</div>
                      ) : (
                        <button className="absolute top-4 right-4 text-gray-400 hover:text-red-500 transition-colors">
                          <Trash2 size={18} />
                        </button>
                      )}
                      
                      <div className="w-12 h-12 rounded-full bg-gray-50 dark:bg-gray-800/50 border border-gray-100 dark:border-gray-700 flex items-center justify-center text-gray-600 dark:text-gray-300 mb-4">
                        {voice.type === 'system' ? <Volume2 size={24} /> : <FileAudio size={24} />}
                      </div>
                      <h3 className="font-semibold text-gray-900 dark:text-white">{voice.name}</h3>
                      <p className="text-sm text-gray-500 dark:text-gray-400 mt-1">
                        {voice.type === 'system' ? 'Standard model optimized for English.' : 'Cloned voice from file "sample_01.wav".'}
                      </p>
                    </div>
                  ))}
                </div>
              </div>
            )}

            {/* SETTINGS VIEW */}
            {activeTab === 'settings' && (
              <div className="max-w-2xl mx-auto bg-white dark:bg-[#161B22] rounded-2xl border border-gray-200 dark:border-gray-800 shadow-sm p-8 transition-colors duration-300">
                <h2 className="text-lg font-semibold mb-6 dark:text-white">Local Model Configuration</h2>
                
                <div className="space-y-6">
                  <div>
                    <label className="block text-sm font-medium text-gray-700 dark:text-gray-300 mb-2">Qwen3 Model Path (.bin / .gguf)</label>
                    <div className="flex gap-2">
                      <input type="text" readOnly value="/models/qwen3-tts-q4_k_m.gguf" className="flex-1 bg-gray-50 dark:bg-[#0E1117] border border-gray-300 dark:border-gray-700 text-gray-900 dark:text-gray-200 text-sm rounded-lg focus:ring-blue-500 focus:border-blue-500 block p-2.5 outline-none" />
                      <button className="px-4 py-2 bg-gray-100 dark:bg-gray-800 text-gray-700 dark:text-gray-300 border border-gray-300 dark:border-gray-700 rounded-lg hover:bg-gray-200 dark:hover:bg-gray-700 transition-colors">Browse</button>
                    </div>
                  </div>

                  <div className="pt-4 border-t border-gray-100 dark:border-gray-800">
                    <label className="block text-sm font-medium text-gray-700 dark:text-gray-300 mb-2">Hardware Acceleration</label>
                    <select className="bg-white dark:bg-[#0E1117] border border-gray-300 dark:border-gray-700 text-gray-900 dark:text-gray-200 text-sm rounded-lg focus:ring-blue-500 focus:border-blue-500 block w-full p-2.5 outline-none">
                      <option>CPU (AVX2)</option>
                      <option>CUDA (NVIDIA GPU)</option>
                      <option>Vulkan (AMD/Intel)</option>
                    </select>
                  </div>
                </div>
              </div>
            )}

          </main>
        </div>
      </div>
    </div>
  );
}

// Helper Component for Navigation
function NavItem({ icon, label, isActive, onClick }) {
  return (
    <button 
      onClick={onClick}
      className={`w-full flex flex-col items-center justify-center p-3 rounded-xl transition-all duration-200 group ${
        isActive 
          ? 'bg-blue-50 text-blue-600 dark:bg-blue-900/20 dark:text-blue-400' 
          : 'text-gray-400 dark:text-gray-500 hover:bg-gray-50 dark:hover:bg-gray-800 hover:text-gray-700 dark:hover:text-gray-300'
      }`}
    >
      <div className={`mb-1 transition-transform ${isActive ? 'scale-110' : 'group-hover:scale-110'}`}>
        {icon}
      </div>
      <span className="text-[10px] font-medium tracking-wide">{label}</span>
    </button>
  );
}
